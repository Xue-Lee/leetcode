class Solution {
    
    public void moveZeroes(int[] nums) {
        if(nums.length == 1)    return;
        int count = 0;
        for(int i=0; i<nums.length-count; i++){
            if(nums[i] == 0){
                for(int j=i;j<nums.length; j++){
                    nums[j]=nums[j+1];
                }
                nums[nums.length-1-count]=0;
                count++;
            }
        }
    }
}
